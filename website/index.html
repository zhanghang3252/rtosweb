5 分钟快速上手

创建你的第一个 MyRTOS 任务

C main.c 
/**
 * MyRTOS 最小示例 — 两个任务交替闪烁 LED
 * 适用于任意开发板，修改 GPIO 即可运行
 */

#include <myrtos.h>
#include <myrtos_device.h>

/* 任务栈大小 */
#define TASK_STACK_SIZE  256

/* 任务句柄 */
static task_handle_t led_task_a;
static task_handle_t led_task_b;

/* LED 闪烁任务 A — 500ms 周期 */
static void led_task_a_entry(void *param) {
    device_t *led = device_find("led_green");
    while (1) {
        device_write(led, 1);   // LED ON
        task_delay_ms(500);
        device_write(led, 0);   // LED OFF
        task_delay_ms(500);
    }
}

/* LED 闪烁任务 B — 1000ms 周期 */
static void led_task_b_entry(void *param) {
    device_t *led = device_find("led_red");
    while (1) {
        device_write(led, 1);   // LED ON
        task_delay_ms(1000);
        device_write(led, 0);   // LED OFF
        task_delay_ms(1000);
    }
}

/* 主函数 — 系统入口 */
int main(void) {
    /* 硬件初始化 */
    board_init();

    /* 创建任务 */
    led_task_a = task_create(
        "led_a",              // 任务名
        led_task_a_entry,    // 入口函数
        NULL,                // 参数
        TASK_STACK_SIZE,     // 栈大小
        3                    // 优先级
    );

    led_task_b = task_create(
        "led_b",
        led_task_b_entry,
        NULL,
        TASK_STACK_SIZE,
        2                    // 较低优先级
    );

    /* 启动调度器 — 不会返回 */
    scheduler_start();

    return 0;
}
